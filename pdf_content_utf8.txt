async function runRecommender() {  
    // 1. เ ร ิ่ มต ้ นระ บ บ  AI ( โ หลด ไ ฟ ล ์ โ มเ ดล ) 
    const isReady = await PSU_AI.init('./');  
    if (!isReady) {  
        console.error("AI failed to load.");  
        return;  
    } 
 
    // 2. ข ้ อ ม ู ล ค  ำตอบขอ ง นั ก ศ ึ ก ษ ำ  ( ต ้ อ ง เ ร ี ย ง ล  ำ ด ั บ ให ้ เ ป ๊ ะตำม ค ู่ม ื อ ด ้ ำนล ่ ำง  จ ำนว น 31 ค ่ำ) 
    const studentAnswers = [  
        4, 5, 2, 3, 5, 1, 2, 3, 4, 5,  
        2, 3, 4, 1, 2, 5, 4, 3, 2, 1,  
        25000, 3.50, 0, 3.45, 3.60, 0, 0, 0, 0, 1, 0  
    ]; 
 
    // 3. ท  ำนำ ย ผ ล ล ั พ ธ ์  
    try { 
        const results = await PSU_AI.predict(studentAnswers);  
        console.log("Prediction Results:", results);  
    } catch (error) {  
        console.error("Prediction failed:", error);  
    } 
} 
 
runRecommender();
